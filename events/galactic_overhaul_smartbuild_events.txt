namespace = galactic_overhaul_smartbuild

# ============================================================================
# SMART BUILD - construction-ship automation.
# v0.9.1: fixed empty queue_action failure effects (now each carries a log -
# empty effect blocks are rejected by the engine and broke the whole queue);
# orbital-ring planet pick uses its own trigger; outpost is_constructing line
# removed. 5 prioritized passes/scan: outpost > relay > gateway > orbital ring
# > habitat. One job at a time (go_smartbuild_blocked_fleet). Log tag:
# GALACTIC_OVERHAUL_SMARTBUILD.
# ============================================================================

# ----- .10 Enrollment sweep -----
country_event = {
	id = galactic_overhaul_smartbuild.10
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { has_policy_flag = go_smartbuild_active }
			every_owned_fleet = {
				limit = { is_ship_size = constructor NOT = { has_fleet_flag = go_smartbuild_ship } }
				set_fleet_flag = go_smartbuild_ship
				fleet_event = { id = galactic_overhaul_smartbuild.1 }
				if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: enrolled constructor [This.GetName]" }
			}
		}
		else = {
			every_owned_fleet = {
				limit = { has_fleet_flag = go_smartbuild_ship }
				remove_fleet_flag = go_smartbuild_ship
				remove_fleet_flag = go_smartbuild_blocked_fleet
				clear_fleet_actions = this
				clear_orders = yes
				if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: released constructor [This.GetName]" }
			}
		}
	}
}

# ----- .11 New-ship enrollment -----
ship_event = {
	id = galactic_overhaul_smartbuild.11
	hide_window = yes
	is_triggered_only = yes
	trigger = { is_ship_class = shipclass_constructor owner = { has_policy_flag = go_smartbuild_active } }
	immediate = {
		fleet = {
			if = {
				limit = { NOT = { has_fleet_flag = go_smartbuild_ship } }
				set_fleet_flag = go_smartbuild_ship
				fleet_event = { id = galactic_overhaul_smartbuild.1 }
				if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: new constructor enrolled [This.GetName]" }
			}
		}
	}
}

		clear_orders = yes
		queue_actions = {
			repeat = {

					# --- PASS 0a: MINING STATION (deposit body, instant build, deficit-first) ---
					find_closest_system = {
						trigger = {
							id = gbsb.1.t.mining
							root = {
								go_smartbuild_fleet_free = { FLEET = root }
								owner = { resource_stockpile_compare = { resource = alloys value >= @go_sb_station_alloys } }
								go_smartbuild_mining_check = { CHECK_SYSTEM = prev }
							}
						}
						found_system = {
							effect = { id = gbsb.1.e.mining_sys if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: [Root.GetName] targeting MINING STATION in [This.GetName]" } }
							move_to = this
							find_closest_planet = {
								trigger = { id = gbsb.1.t.mining_p has_deposit_for = shipclass_mining_station }
								found_planet = {
									orbit_planet = this
									effect = {
										id = gbsb.1.e.mining_p
										create_mining_station = { owner = root.owner }
										root = { owner = { add_resource = { alloys = @go_sb_station_alloys_neg } } }
										if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: [Root.GetName] BUILT MINING STATION at [This.GetName]" }
									}
								}
								failed = { effect = { id = gbsb.1.e.mining_f if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: [Root.GetName] mining: no deposit body" } } }
							}
						}
						failed = { effect = { id = gbsb.1.e.mining_x if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: [Root.GetName] no mining target" } } }
					}

						}

				find_closest_system = {
					trigger = {
						id = gbsb.1.t.outpost

							go_smartbuild_fleet_free = { FLEET = root }

							go_smartbuild_outpost_check = { CHECK_SYSTEM = prev }
						}
					}

						effect = {
							id = gbsb.1.e.outpost_sys

# ----- .2 Build OUTPOST -----
planet_event = {
	id = galactic_overhaul_smartbuild.2
	hide_window = yes
	is_triggered_only = yes
	trigger = { any_fleet_in_orbit = { has_fleet_flag = go_smartbuild_ship has_fleet_flag = go_smartbuild_blocked_fleet } }
	immediate = {
		random_fleet_in_orbit = {
			limit = { has_fleet_flag = go_smartbuild_ship has_fleet_flag = go_smartbuild_blocked_fleet }
			owner = { save_event_target_as = gbsb_builder }
			if = {
				limit = {
					solar_system = { NOR = { exists = space_owner } }
					event_target:gbsb_builder = {
						resource_stockpile_compare = { resource = influence value >= @go_sb_outpost_influence }
						resource_stockpile_compare = { resource = alloys value >= @go_sb_outpost_alloys }
					}
				}
				solar_system = { create_starbase = { size = starbase_outpost owner = event_target:gbsb_builder } }
				event_target:gbsb_builder = {
					add_resource = { influence = @go_sb_outpost_influence_neg }
					add_resource = { alloys = @go_sb_outpost_alloys_neg }
				}
				if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: [This.GetName] CLAIMED OUTPOST" }
			}
			else = { if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_SMARTBUILD: [This.GetName] outpost aborted" } }
			remove_fleet_flag = go_smartbuild_blocked_fleet
		}
	}
}
