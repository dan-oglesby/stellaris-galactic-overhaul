# Button effects for the Galactic Bank panel (interface/go_bank_window.gui).
#
# SCOPE CONTRACT (documented by Paradox in vanilla common/button_effects/example.txt):
#     This/Root = the currently SELECTED object (planet, fleet, system...) or the
#                 player country if nothing is selected. UNRELIABLE - never use it.
#     From      = the player country. ALWAYS. Use this.
# Hyperlane Master's working panel relies on exactly this, so `From` was never the bug.
#
# WHY THE BUTTONS DID NOTHING (v0.19.2): the affordability check lived in `allow`.
# When an `allow` block is false the engine renders the button inert and swallows
# the click with NO popup, NO error and NO log line - indistinguishable from a dead
# button. The guard now lives inside `effect`, so a click always registers and logs
# why it did or did not transfer. If deposits log BLOCKED, the trade stockpile is
# genuinely too low (Trade Value is largely a monthly flow, so it may not bank up
# the way energy/minerals do - the log will tell us definitively).
#
# NOTE: the buttons no longer re-fire go_bank.1. Re-firing does NOT repaint an open
# custom_gui window - it just queues a second one. The readouts now self-refresh
# because go_bank_script_values.txt reads live resources on each GUI redraw.

# no-op used by the transparent text readouts and the close button
go_bank_readout_noop = {
	potential = { always = yes }
	allow     = { always = yes }
	effect    = { }
}

# ---------- DEPOSIT (trade -> bank) ----------
go_bank_deposit_1000 = {
	potential = { always = yes }
	allow     = { always = yes }
	effect = {
		hidden_effect = {
			From = {
				if = {
					limit = { resource_stockpile_compare = { resource = trade value >= 1000 } }
					go_bank_do_deposit = { AMOUNT = 1000 }
				}
				else = {
					if = { limit = { go_debug_on = yes } log = "GALACTIC_OVERHAUL_BANK: panel DEPOSIT 1000 BLOCKED - trade stockpile below 1000" }
				}
			}
		}
		custom_tooltip = "GO_BANK_DEP_1K"
	}
}
