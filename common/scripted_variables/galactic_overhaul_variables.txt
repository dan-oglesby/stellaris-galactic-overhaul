# ============================================================================
# Galactic Bank - tunable constants (scripted variables, substituted at parse time)
# ============================================================================

# Base monthly interest rates by policy tier.
@go_rate_conservative = 0.0025
@go_rate_standard      = 0.005
@go_rate_aggressive    = 0.01
# Extra flat rate for the Galactic Bank host empire.
@go_rate_host_bonus    = 0.005

# ---- Dynamic activity from price signals (SELF-CALIBRATING) ----
# Activity is derived from the galactic mineral price's own recent behaviour:
# we track an exponentially-weighted recent mean and its mean-absolute-deviation
# (MAD = typical wobble). "Activity" = how many MADs the current price sits from
# that mean, in EITHER direction. A sharp move (up or down) vs recent calm = high
# activity = higher interest. No external baseline needed.

# EMA smoothing. Higher = shorter memory / more reactive. 0.15 ~ a few-month memory.
@go_alpha              = 0.15
# Noise floor: deviations smaller than this many MADs count as nothing.
@go_z_min              = 1.0
# Ramp width: MADs of deviation (beyond the floor) over which activity goes 0 -> max.
@go_z_range            = 2.0
# Maximum activity bonus. 1.0 = up to +100% (the rate can double at peak activity).
@go_activity_max       = 1.0
# Floor on MAD to avoid divide-by-zero on a perfectly flat / brand-new market.
@go_mad_epsilon        = 0.01

# ---- Savings dampening (counterweight to high activity) ----
# As reserves grow the rate falls. dampening = reserves / scale, capped at maxdamp.
@go_savings_scale      = 200000
@go_savings_maxdamp    = 0.75

# --- Bank risk of loss (tiers above Conservative carry downside) ---
@go_bank_loss_std = 0.05   # Standard: lose this fraction of banked credits on a bad month (~8% monthly chance)
@go_bank_loss_agg = 0.10   # Aggressive: lose this fraction of banked credits on a bad month (~15% monthly chance)
